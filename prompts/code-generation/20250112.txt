Request
จงสร้าง function ที่สร้างด้วยภาษา javascript เพื่อแปลงข้อความ (string) ใดๆ ให้เป็นข้อความแบบ title case
กรณีค่าที่ได้รับเป็น null หรือ undefined หรือ ค่าว่างหรือ มีแต่ช่องว่างจะได้ผลลัพธ์ เป็นค่าว่าง
function นี้ควรสามารถจัดการกับตัวเลข และ อักขระพิเศษได้

ใน function มีเอกสารการอธิบาย และ comment อธิบายใน code ไปเลย ไม่ต้องแยกการอธิบายออกมาเป็นหัวข้อต่างหาก
มีการแสดงตัวอย่างการใช้งาน กรณีปกติและกรณีใส่ตัวเลขผิด
รูปแบบของผลลัพธ์ที่ต้องการคือ คำนำ, วิเคราะห์, ส่วนของโค้ดและตัวอย่าง , สรุป


Response
เราสามารถสร้างฟังก์ชันในภาษา JavaScript เพื่อจัดระเบียบข้อมูลให้ดูสวยงามขึ้น โดยข้อความแบบ title case จะประกอบไปด้วยตัวอักษรตัวแรกของแต่ละคำที่เป็นตัวพิมพ์ใหญ่ และตัวอักษรที่เหลือเป็นตัวพิมพ์เล็ก การสร้างฟังก์ชันที่สามารถแปลงข้อความให้เป็นแบบนี้ได้อย่างมีประสิทธิภาพจึงเป็นเรื่องที่มีประโยชน์ โดยเฉพาะอย่างยิ่งเมื่อสามารถจัดการกับข้อมูลที่เป็น null, undefined หรือมีแต่ช่องว่างได้ถูกต้อง รวมถึงข้อควาที่มีตัวอักษรและอักขระพิเศษต่างๆ



วิเคราะห์: 
การแปลงข้อความเป็นแบบ Title Case คือการทำให้ตัวอักษรตัวแรกของแต่ละคำเป็นตัวพิมพ์ใหญ่และตัวอักษรที่เหลือเป็นตัวพิมพ์เล็ก โดยต้องพิจารณา
ข้อความปกติสามารถแปลงเป็น title case ได้ถูกต้อง
ข้อความมีการเว้นวรรคทั้งก่อนหน้าและหลังข้อความ ต้องสามารถจัดการแปลงข้อความได้ถูกต้อง
ตัวเลข หรืออักขระพิเศษ ที่ไม่สามารถแปลงได้โดยตรงจะไม่มีการเปลี่ยนแปลง
การจัดการกับค่าที่เป็น null, undefined หรือค่าว่าง เพราะไม่สามารถนำมาแปลงได้โดยตรง

จากการวิเคราะห์สามารถสร้าง code ได้ดังนี้

/**
 * แปลงข้อความที่ได้รับมาให้อยู่ในรูปแบบ title case
 * กรณีที่ค่าที่ได้รับมาเป็น null, undefined หรือ มีแต่ช่องว่างจะให้ผลลัพธ์เป็นค่าว่าง
 * ตัวเลขและอักขระพิเศษที่อยู่ในข้อความจะไม่ถูกแปลง
 *
 * @param {string} text - ข้อความที่ต้องการแปลงให้อยู่ในรูปแบบ title case
 * @return {string} ข้อความที่อยู่ในรูปแบบ title case
 */
function toTitleCase(text) {
  // ตรวจสอบกรณีข้อความที่รับมาเป็น null, undefined หรือ มีแต่ช่องว่างจะให้ผลลัพธ์เป็นค่าว่าง ทันที
  if (!text || text.trim() === '') {
    return '';
  }

  // แยกคำแต่ละคำด้วยการใช้ช่องว่าง
  return text.split(' ').map(word => {
    // แยกตัวอักษรตัวแรกเพื่อแปลงเป็นตัวพิมพ์ใหญ่ ตัวอักษรที่เหลือเป็นตัวพิมพ์เล็ก แล้วรวมกันเป็นคำ
    return word.charAt(0).toUpperCase() + word.slice(1).toLowerCase();
  }).join(' '); // รวมคำที่แปลงแล้วกลับ
}

ตัวอย่างการใช้งาน

// การใช้งานกรณีต่างๆ
console.log(toTitleCase("hello world")); // ผลลัพธ์: "Hello World"
console.log(toTitleCase("javaScript is fun!")); // ผลลัพธ์: "Javascript Is Fun!"
console.log(toTitleCase("  ")); // ผลลัพธ์: ""
console.log(toTitleCase(null)); // ผลลัพธ์: ""
console.log(toTitleCase("  multiple   spaces  ")); // ผลลัพธ์: "Multiple Spaces"
console.log(toTitleCase("123 testing")); // ผลลัพธ์: "123 Testing"
console.log(toTitleCase("!special*chars#")); // ผลลัพธ์: "!special*chars#"

สรุป: 
ฟังก์ชัน toTitleCase ที่สร้างขึ้นด้วย javascript สามารถแปลงข้อความใดๆ ให้เป็นแบบ Title Case ได้อย่างมีประสิทธิภาพ โดยครอบคลุมกรณีที่ข้อความเป็น null, undefined, หรือว่าง และสามารถจัดการกับตัวเลขและอักขระพิเศษได้อย่างถูกต้อง การใช้งานฟังก์ชันนี้จะช่วยให้การจัดรูปแบบข้อความเป็นไปอย่างสะดวกและง่ายดายมากยิ่งขึ้น
